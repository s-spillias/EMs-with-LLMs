 template<class Type>                         
 Type objective_function<Type>::operator() ()

   DATA_VECTOR(cots_dat);    // Starfish

 Log-standard deviation for observation error 
 ensures sigma>0

   Type eps = Type(1e-8);  // Small constant  
 for numerical stability                      
                                              
   int n = cots_dat.size();   // Number of    
 observations

 negative log-likelihood accumulator          
                                              
   // 4. Compute predictions with smooth      
 transitions:                                 
   vector<Type> slow_pred(n); // Predicted    
 slow-growing coral cover (%)                 
   vector<Type> fast_pred(n); // Predicted    
 fast-growing coral cover (%)                 
   for (int i = 0; i < n; i++){

 prediction using a logistic-type smooth      
 transition

 Eq.3 for fast coral

   REPORT(slow_pred);   // slow_pred: predict

   REPORT(fast_pred);   // fast_pred: predict 
 fast-growing coral cover (%)                 
   REPORT(sigma);       // sigma: effective   
 observation error standard deviation         
                                              
   return nll;                                
 }
