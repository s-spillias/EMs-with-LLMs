 #include <TMB.hpp>                              
                                                 
 // Template Model Builder (TMB) model for       
 Crown-of-Thorns starfish and coral dynamics     
 // 1. MODEL EQUATIONS:                          
 //    (1) slow_pred: Predicted slow-growing cor 
 cover equals the observed value (placeholder    
 model).                                         
 //    (2) fast_pred: Predicted fast-growing cor 
 cover equals the observed value (placeholder    
 model).                                         
 //    (3) cots_pred: Predicted crown-of-thorns  
 starfish abundance is modeled as an exponential 
 function of coral covers.                       
 // 2. PARAMETERS:                               
 //    - log_sd: Log of the observation standard 
 deviation to ensure positive values (units:     
 log-scale).                                     
 //    - alpha: Predation rate on slow-growing   
 corals (Faviidae and Porites). Unitless scaling 
 parameter.                                      
 //    - beta: Predation rate on fast-growing    
 corals (Acropora spp.). Unitless scaling        
 parameter.                                      
 //                                              
 // IMPORTANT CONSIDERATIONS:                    
 //    * Numerical stability is ensured via a    
 small constant (epsilon = 1e-8).                
 //    * Smooth transitions are applied using    
 divisions that include epsilon.                 
 //    * All observations are included in the    
 likelihood with lognormal distributions after   
 log-transformation.                             
 //    * Smooth penalties (Gaussian priors) are  
 applied to bound parameters within meaningful   
 ranges.                                         
 template<class Type>                            
 Type objective_function<Type>::operator() ()    
 {                                               
     using namespace density;                    
                                                 
     // DATA                                     
     DATA_VECTOR(slow_dat);   // Observed        
 slow-growing coral cover (%)                    
     DATA_VECTOR(fast_dat);   // Observed        
 fast-growing coral cover (%)                    
     DATA_VECTOR(cots_dat);   // Observed        
 crown-of-thorns starfish abundance              
 (individuals/m^2)                               
                                                 
     // PARAMETERS                               
     PARAMETER(log_sd);       // Log observation 
 standard deviation (log-scale)                  
     PARAMETER(alpha);        // Predation rate  
 slow-growing corals; typical range approx. -3 t 
 3                                               
     PARAMETER(beta);         // Predation rate  
 fast-growing corals; typical range approx. -3 t 
 3                                               
                                                 
     // Small constant for numerical stability   
     Type epsilon = Type(1e-8);                  
                                                 
     // Initialize negative log-likelihood       
     Type nll = 0.0;                             
                                                 
     // Numbered list of equations:              
     // (1) slow_pred: In this placeholder,      
 prediction equals observed slow coral cover.    
     // (2) fast_pred: In this placeholder,      
 prediction equals observed fast coral cover.    
     // (3) cots_pred: Predicted starfish        
 abundance via an exponential function including 
 smooth transitions.                             
     vector<Type> slow_pred = slow_dat;   // (1) 
 slow-growing coral cover prediction (%)         
     vector<Type> fast_pred = fast_dat;   // (2) 
 fast-growing coral cover prediction (%)         
     vector<Type> cots_pred(slow_dat.size()); // 
 (3) crown-of-thorns starfish abundance predicti 
 (individuals/m^2)                               
                                                 
     for(int i = 0; i < slow_dat.size(); i++){   
         // Using smooth transitions with epsilo 
 to avoid division by zero.                      
         cots_pred(i) = exp( (alpha *            
 slow_dat(i))/(slow_dat(i) + epsilon) +          
                              (beta  *           
 fast_dat(i))/(fast_dat(i) + epsilon) );         
     }                                           
                                                 
     // Likelihood calculation using lognormal   
 distributions with fixed minimum standard       
 deviations.                                     
     Type sd = exp(log_sd);  // Ensure standard  
 deviation is positive                           
     for(int i = 0; i < slow_dat.size(); i++){   
         // Using log-transformation to handle   
 data spanning multiple orders of magnitude.     
         nll -= dnorm(log(slow_dat(i) + epsilon) 
 log(slow_pred(i) + epsilon), sd, true);  // slo 
 coral cover                                     
         nll -= dnorm(log(fast_dat(i) + epsilon) 
 log(fast_pred(i) + epsilon), sd, true);  // fas 
 coral cover                                     
         nll -= dnorm(log(cots_dat(i) + epsilon) 
 log(cots_pred(i) + epsilon), sd, true);  //     
 crown-of-thorns starfish                        
     }                                           
                                                 
     // Smooth penalties to bound parameters     
 (using Gaussian priors)                         
     nll += dnorm(alpha, Type(0), Type(1), true) 
 // Prior for alpha                              
     nll += dnorm(beta, Type(0), Type(1), true); 
 // Prior for beta                               
                                                 
     // REPORT predictions for further analysis  
 (suffixed with _pred)                           
     REPORT(slow_pred); // Reporting slow_pred:  
 predicted slow-growing coral cover (%)          
     REPORT(fast_pred); // Reporting fast_pred:  
 predicted fast-growing coral cover (%)          
     REPORT(cots_pred); // Reporting cots_pred:  
 predicted crown-of-thorns starfish abundance    
 (individuals/m^2)                               
                                                 
     return nll;                                 
 }
